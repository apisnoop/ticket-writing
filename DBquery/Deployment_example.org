   select endpoint,
          unnest(tests) as test
     from endpoint_coverage
    where endpoint like '%Deployment%'
       and endpoint not like '%alpha%'
       and endpoint not like '%beta%'
     -- and release = '1.19.0'
     group by test, endpoint, level, category;
